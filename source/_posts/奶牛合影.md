const int maxn = 6e5+100;

int s[maxn];
int getminpos(int len) {
    int l = len*2;
    for(int i = len; i < l; ++i){
        s[i] = s[i-len];
    }
    s[l] = 0;
    int i = 0, j = 1, k = 0;
    while(i < len && j < len && k < len) {
        if(s[i+k] == s[j+k]) k++;
        else if(s[i+k] < s[j+k]) {
            j = max(i+1,j+k+1);
            k = 0;
        } else i = max(j+1,i+k+1), k = 0;
    }
    if(i < len) return i;
    return j;
}
